#include <iostream>

using namespace std;
class Interv{
    double inf;
    double sup;
    public:
        Interv(double a,double b):inf{a},sup{b}{}
        //Interv(double a=1,double b=0):inf{a},sup{b}{} fait attention ecrit pas cela provoque la conflit avec la constructeur vide

        Interv(double a):inf{a},sup{a}{};
        Interv():sup{0},inf{1}{};
            

        
        int appartient(double v){
            if(this->sup>=v && this->inf<=v){
                return 1;
            }
            return 0;
        }

};
