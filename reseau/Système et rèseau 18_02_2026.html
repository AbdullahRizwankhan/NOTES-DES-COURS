point important:-

  • ­Un processus peut contenir plusieurs threads.**

L’API PTHREADS:-
Standardisé à la fin des annéed 90.

pthread_create():- créé un nouveau thread.
void arg: pour passer des arguments à la fonction.

pthread_join:- equiavalent de wait en manière flou.

phthread_detach():- c’est pour nétooyer automiquement après la sortie.

#include <stdlib.h>
#include <fcntl.h>
#include <unistd.h>
#include <stdio.h>
#include <pthread.h>
#include <errno.h>
#include <string.h>

static int toInc = 0;
pthread_mutex_t mutex = PTHREAD_MUTEX_INITIALIZER;

static void* threadF(void* arg){

    int* loops = (int *) arg;
    int i, temp; 

    for (i = 0; i < *loops; i++){
        pthread_mutex_lock(&mutex);
        temp = toInc;
        temp++;
        toInc = temp;
        pthread_mutex_unlock(&mutex);
    }
    
    return NULL;

}

int main(int argc, char *argv[]){

    int n;
    if (argc > 1){
        n = atoi(argv[1]);
    }
    else{
        printf("Wrong input\n");
        exit(EXIT_FAILURE);
    }
    pthread_mutex_init(&mutex,NULL);

    pthread_t t1, t2;

    int s = pthread_create(&t1, NULL, threadF, &n);
    if (s!=0){
        printf("Error in creating t1: %s", strerror(s));
        exit(EXIT_FAILURE);
    }    

    s = pthread_create(&t2, NULL, threadF, &n);
    if (s!=0){
        printf("Error in creating t2: %s", strerror(s));
        exit(EXIT_FAILURE);
    }

    //wait for the threads to finish
    s = pthread_join(t1, NULL); 
    if (s!=0){
        printf("Error in joining t1: %s", strerror(s));
        exit(EXIT_FAILURE);
    }
    s = pthread_join(t2, NULL); 
    if (s!=0){
        printf("Error in joining t2: %s", strerror(s));
        exit(EXIT_FAILURE);
    }
    pthread_mutex_destroy(&mutex);


    printf("Final value: %d\n",toInc);
    exit(EXIT_SUCCESS);
}

    typedef struct node {
        struct node * next;
    } node_t;
    static int toInc = 0;
    static pthread_mutex_t mtx = PTHREAD_MUTEX_INITIALIZER;
    static void* threadF(void* arg){
        node_t* head = (node_t *) arg;
        pthread_mutex_lock(&mtx);

        if(head!=NULL && head->next!=NULL){
                node_t* next_node = head->next;
                head->next=next_node->next;

                free(next_node);
        }
        //deletes the next-after-the-head node and link the head to the next-to-next
        //head-next-nnext --> head-nnext
        pthread_mutex_unlock(&mtx);


        return NULL;
    }
